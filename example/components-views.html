StateView

When included allows templates to be easily included in pages and configured during state definition.

Definition

Templates can be either functions that generate responses directly or HTML fragments included through ng-include

HTML Fragment

Function

angular.module('myApp', ['angular-state-router'])
.config(function($stateProvider) {

$stateProvider

// Define states
.state('landing', {
  url: '/'
})

// Set initialization location; optionally
.init('landing');

});

Usage

Access application state through $state service

angular.module('myApp')
.controller('MyController', function($scope, $state) {

// Listen to events
$state.on('change:complete', function() {

// Get the current state
var current = $state.current();

console.log(current.name);
});

$scope.onClickMarketingCampaign = function() {

// Request changes
$state.change('catalogs.products.items', { id: 'e239' });

};

});
